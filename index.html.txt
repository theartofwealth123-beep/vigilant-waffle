    function updateTeams() {
        const league = document.getElementById('league').value;
        const home = document.getElementById('homeTeam');
        const away = document.getElementById('awayTeam');
        home.innerHTML = '<option value="">Select Home Team</option>';
        away.innerHTML = '<option value="">Select Away Team</option>';
        leagues[league].forEach(t => {
            home.innerHTML += `<option value="${t}">${t}</option>`;
            away.innerHTML += `<option value="${t}">${t}</option>`;
        });

    function genStats(team, league) {
        const seed = team.split('').reduce((a, c) => a + c.charCodeAt(0), 0);
        const r = (min, max, o = 0) => {
            const v = ((seed + o) * 9301 + 49297) % 233280;
            return min + (v / 233280) * (max - min);
        };
        const homeAdv = league === 'NFL' ? 2.8 : league === 'NBA' ? 3.5 : league === 'MLB' ? 0.3 : league === 'NCAAF' ? 3.5 : 4.2;
        const paceR = league === 'NBA' ? [96, 104] : league === 'NCAAB' ? [65, 73] : [60, 72];
        return {
            offRating: r(15, 40, 1), defRating: r(15, 35, 2), ppg: r(18, 35, 3), papg: r(17, 30, 4),
            winPct: r(0.3, 0.75, 5), sos: r(0.4, 0.65, 6), toDiff: r(-12, 15, 7),
            homeAdv, extra1: r(0.4, 0.7, 8), extra2: r(0.35, 0.65, 9), pace: r(paceR[0], paceR[1], 10)
        };
    }

    function predictScore() {
        const league = document.getElementById('league').value;
        const homeTeam = document.getElementById('homeTeam').value;
        const awayTeam = document.getElementById('awayTeam').value;
        if (!homeTeam || !awayTeam) return alert('Select both teams');
        if (homeTeam === awayTeam) return alert('Select different teams');

        const h = genStats(homeTeam, league);
        const a = genStats(awayTeam, league);
        const pFH = h.pace / ((h.pace + a.pace) / 2);
        const pFA = a.pace / ((h.pace + a.pace) / 2);
        const sosH = 1 + (h.sos - 0.5) * 0.4;
        const sosA = 1 + (a.sos - 0.5) * 0.4;

        const s1h = (h.offRating - a.defRating) * pFH + h.homeAdv;
        const s1a = (a.offRating - h.defRating) * pFA;
        const s2h = (h.ppg * sosH * (1 + h.winPct * 0.2) * pFH) + h.homeAdv;
        const s2a = (a.ppg * sosA * (1 + a.winPct * 0.2) * pFA);
        const s3h = ((h.ppg + (100 - a.papg)) / 2) * pFH + h.homeAdv;
        const s3a = ((a.ppg + (100 - h.papg)) / 2) * pFA;
        const s4h = h.ppg * (1 + h.extra1 * 0.3) * (1 + h.extra2 * 0.2) * sosH * pFH + h.homeAdv - (h.toDiff * 0.2);
        const s4a = a.ppg * (1 + a.extra1 * 0.3) * (1 + a.extra2 * 0.2) * sosA * pFA - (a.toDiff * 0.2);

        const hScore = Math.round((s1h * 0.3 + s2h * 0.3 + s3h * 0.2 + s4h * 0.2) * 10) / 10;
        const aScore = Math.round((s1a * 0.3 + s2a * 0.3 + s3a * 0.2 + s4a * 0.2) * 10) / 10;
        const conf = Math.abs(hScore - aScore) > 10 ? 'High' : Math.abs(hScore - aScore) > 5 ? 'Medium' : 'Low';

        document.getElementById('confidence').textContent = conf + ' Confidence';
        document.getElementById('homeTeamName').textContent = homeTeam;
        document.getElementById('awayTeamName').textContent = awayTeam;
        document.getElementById('homeScore').textContent = hScore;
        document.getElementById('awayScore').textContent = aScore;
        document.getElementById('winner').textContent = hScore > aScore ? homeTeam : awayTeam;
        document.getElementById('spread').textContent = 'Spread: ' + Math.abs(hScore - aScore).toFixed(1);

        document.getElementById('homeStatsTitle').textContent = homeTeam + ' Stats';
